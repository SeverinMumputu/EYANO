1 Speech-to-Text & Transmission

La voix du pratiquant est convertie localement par l'API Web Speech en texte JSON de manière instantanée, puis transmise au endpoint backend `/api/v1/dialogue/process`.

2 Parser Déterministe (Avatar Core)

Le serveur Node exécute un algorithme de corrélation basé sur des règles grammaticales, compare les mots clés avec la base de connaissances locale MySQL, et détermine le contexte (mémoire).

3 Synthèse Vocale (Google TTS)

Le texte espagnol de réponse sélectionné est envoyé au moteur Google TTS pour générer l'audio d'appel qui est joué immédiatement sur l'interface du client, mimant un flux d'appel téléphonique.

/* Controller de Dialogue Vocal pour l'Avatar Core */
const db = require('../config/db');

exports.processVocalInput = async (req, res) => {
    const { transcription, context, activeAvatar } = req.body;
    
    if (!transcription) {
        return res.status(400).json({ error: "Aucune transcription fournie" });
    }

    try {
        const cleanedText = transcription.toLowerCase().trim();
        
        // 1. Recherche de correspondance de mots clés dans MySQL
        const query = `
            SELECT * FROM eyano_rules 
            WHERE context = ? AND MATCH(keywords) AGAINST(? IN BOOLEAN MODE)
            LIMIT 1
        `;
        
        const [rules] = await db.execute(query, [context, cleanedText]);
        let selectedResponse = "";
        let matchedRuleId = null;
        
        if (rules.length > 0) {
            selectedResponse = rules[0].response_text;
            matchedRuleId = rules[0].id;
        } else {
            // Règle de repli déterministe par défaut selon le contexte
            const [fallback] = await db.execute(
                `SELECT response_text FROM eyano_rules WHERE context = ? AND is_fallback = 1 LIMIT 1`,
                [context]
            );
            selectedResponse = fallback.length > 0 ? fallback[0].response_text : "Lo siento, ¿podrías repetirlo de otra manera?";
        }

        // 2. Mise à jour de la table de mémoire d'historique conversationnel
        await db.execute(
            `INSERT INTO eyano_memory_state (context_id, user_phrase, avatar_response, matched_rule) VALUES (?, ?, ?, ?)`,
            [context, transcription, selectedResponse, matchedRuleId]
        );

        // 3. Retour de la structure de dialogue à l'interface client pour Google Text-to-Speech
        return res.status(200).json({
            success: true,
            response_text: selectedResponse,
            rule_id: matchedRuleId,
            details: rules.length > 0 ? rules[0].grammar_focus : "Phrase standard de rechange"
        });

    } catch (error) {
        console.error("Erreur d'analyse d'avatar logic :", error);
        return res.status(500).json({ error: "Erreur interne de traitement Eyano Core" });
    }
};
-- Création de la Base de Données d'Eyano
CREATE DATABASE IF NOT EXISTS eyano_local_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE eyano_local_db;

-- Table de configuration de la personnalité de l'avatar
CREATE TABLE IF NOT EXISTS eyano_avatar_config (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    voice_key VARCHAR(50) NOT NULL,
    accent VARCHAR(10) DEFAULT 'es-ES',
    personality_description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Table des règles déterministes de dialogue
CREATE TABLE IF NOT EXISTS eyano_rules (
    id INT AUTO_INCREMENT PRIMARY KEY,
    context VARCHAR(50) NOT NULL,
    keywords TEXT NOT NULL,
    response_text TEXT NOT NULL,
    grammar_focus TEXT,
    is_fallback TINYINT(1) DEFAULT 0,
    FULLTEXT INDEX idx_keywords (keywords)
) ENGINE=InnoDB;

-- Table de mémoire d'état conversationnel d'appel
CREATE TABLE IF NOT EXISTS eyano_memory_state (
    id INT AUTO_INCREMENT PRIMARY KEY,
    context_id VARCHAR(50) NOT NULL,
    user_phrase TEXT NOT NULL,
    avatar_response TEXT NOT NULL,
    matched_rule INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (matched_rule) REFERENCES eyano_rules(id) ON DELETE SET NULL
);
